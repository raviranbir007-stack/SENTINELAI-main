// Security Posture Logic let lastPostureResp = null; async function fetchSecurityPosture() { try { const resp = await apiRequest('/dashboard/security-posture'); lastPostureResp = resp; const statusDiv = document.getElementById('securityPostureStatus'); if (!resp || !resp.summary) { statusDiv.textContent = '--'; statusDiv.onclick = null; statusDiv.style.textDecoration = 'none'; statusDiv.style.color = ''; statusDiv.style.fontWeight = ''; statusDiv.title = ''; return; } const { critical_findings, high_findings, total_findings } = resp.summary; if (critical_findings > 0 || high_findings > 0) { statusDiv.innerHTML = '🛡️ '; statusDiv.title = 'Click to review and fix findings'; statusDiv.style.cursor = 'pointer'; const btn = document.getElementById('securityPostureWarningBtn'); if (btn) { btn.onclick = function() { showSecurityPostureModal(lastPostureResp); }; } statusDiv.onclick = null; } else { statusDiv.innerHTML = '🛡️ Normal'; statusDiv.title = ''; statusDiv.style.cursor = 'default'; statusDiv.onclick = null; } } catch (e) { const statusDiv = document.getElementById('securityPostureStatus'); statusDiv.textContent = '--'; statusDiv.onclick = null; statusDiv.style.textDecoration = 'none'; statusDiv.style.color = ''; statusDiv.style.fontWeight = ''; statusDiv.title = ''; } } function showSecurityPostureModal(posture) { const modal = document.getElementById('securityPostureModal'); const findingsDiv = document.getElementById('securityPostureFindings'); const findings = posture.report?.findings || {}; let findingsList = ''; for (const [cat, items] of Object.entries(findings)) { if (!Array.isArray(items) || items.length === 0) continue; findingsList += `

${cat.charAt(0).toUpperCase() + cat.slice(1)}

    `; for (const item of items) { findingsList += `
  • ${item.title || item.name}: ${item.description || ''}
    Severity: ${item.severity || ''}${item.remediation ? `
    Fix: ${item.remediation}` : ''}
    • `; } findingsList += '
'; } if (!findingsList) findingsList = '
No detailed findings available.
'; findingsDiv.innerHTML = findingsList; modal.style.display = 'flex'; } document.getElementById('fixAllPostureBtn').onclick = async function() { this.disabled = true; this.textContent = 'Fixing...'; try { const resp = await apiRequest('/dashboard/fix-security-posture', { method: 'POST' }); showToast(resp?.message || 'Remediation attempted. Please re-scan to verify.', resp?.success ? 'success' : 'warning'); document.getElementById('securityPostureModal').style.display = 'none'; fetchSecurityPosture(); } catch (e) { showToast('Failed to fix posture issues: ' + (e.message || e), 'error'); } finally { this.disabled = false; this.textContent = 'Fix All Automatically'; } }; // Call on load and every 30s fetchSecurityPosture(); setInterval(fetchSecurityPosture, 30000);
